#include <stdio.h>
int main(){
	//declaration
	int a[]={1,2,3,4,5}, flag=0, temp=0;
	//running an infinite loop
	for(;;){
		flag=0; //reinitializing the flag variable for looping check
		for(int i=0; i<5; i++){
			if(a[i]<a[i+1]){
				temp=a[i];
				a[i]=a[i+1];
				a[i+1]=temp;
				flag++;//value of flag increases if changes took place
			}
			
		}
		//checking for the boolean value
		if(flag==0){
				break;// breaks the infinite loop if no changes were made
			}

	}
	//traversing the array
	for(int i=0; i<5; i++){
		printf("%d ", a[i]);
	}
}
